def create
        @user = User.new(user_params)
        if @user.save
                if send_activation_emails?
                        @user.send_activation_email
flash[:info] = "Please check your email to activate your account."
redirect_to root_url
                else
                        @user.activate
                        log_in(@user)
flash[:success] = "Welcome to #{competition_name}!"
redirect_to @user
                end
        else
                render 'new'
        end
end
