def create
        @hint = Hint.new(hint_params)
        @problem = Problem.find(params[:hint][:problem_id])
        if @hint.save
                @problem.add(@hint.id)
                flash[:success] = "Hint saved!"
                redirect_to @problem
        else
                render 'new'
        end
end
